30      public String selectStar(String table, String query){        
31          return query != null? "SELECT * FROM " + table + " WHERE " + query : "SELECT * FROM " + table;
32      }
33  
34      /**
35       * Produces a parametrized  AND query.
36       * Example:
37       * <pre>
38       * String sql = dialect.selectStarParametrized("people", "name", "ssn", "dob");
39       * //generates:
40       * //SELECT * FROM people WHERE name = ? AND ssn = ? AND dob = ?
41       * </pre>
42       *
43       *
44       * @param table name of table
45       * @param parameters list of parameter names
46       * @return something like: "select * from table_name where name = ? and last_name = ? ..."
47       */
48      public String selectStarParametrized(String table, String ... parameters) {
49          String sql = "SELECT * FROM " + table + " WHERE ";
50  
51          for(String parameter:parameters){
52              sql += parameter + " = ? AND ";
53          }
54          return sql.substring(0, sql.length() - 5);//remove last comma
55      }
56  
57      public String createParametrizedInsert(MetaModel mm){
58          List<String> attributes = mm.getAttributeNamesSkip("record_version", mm.getIdName());
59          String query = "INSERT INTO " + mm.getTableName() + " (" + Util.join(attributes, ", ");
60          query += mm.getIdGeneratorCode()!= null ? ", " + mm.getIdName() :"";
61          query += mm.isVersioned()? ", " + "record_version" :"";
62          query += ") VALUES ("+ getQuestions(attributes.size());
63          query += mm.getIdGeneratorCode() != null ? ", " + mm.getIdGeneratorCode() :"";
64          query += mm.isVersioned()? ", " + 1 :"";
65          query +=")";
66  
67  
68          return query; 
69      }
70  
71      private String getQuestions(int count){
72          String [] questions = new String[count];
73          for(int i = 0; i < count; i++){
74              questions[i] = "?";
75          }
76          return Util.join(questions, ", ");
77      }

79     public String formSelect(String tableName, String subQuery, List<String> orderBys, long limit, long offset) {
80  
81          String fullQuery = "SELECT * FROM " + tableName;
82  
83          if(!Util.blank(subQuery)){
84              String where = " WHERE ";
85              //this is only to support findFirst("order by..."), nightneed to revisit later
86              if(!subQuery.toLowerCase().trim().startsWith("order") && !subQuery.toLowerCase().trim().startsWith("group")){
87                  fullQuery += where;
88              }
89              fullQuery += " " + subQuery;
90          }
91  
92  
93          if(orderBys.size() != 0){
94              fullQuery += " ORDER BY " + Util.join(orderBys, ", ");
95          }
96  
97          return fullQuery;
98      }
