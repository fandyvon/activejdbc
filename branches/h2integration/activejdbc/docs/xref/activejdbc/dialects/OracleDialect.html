18  package activejdbc.dialects;
19  
20  import javalite.common.Util;
21  
22  import java.util.List;
23  
24  /**
25   * @author Igor Polevoy
26   */
27  public class OracleDialect extends DefaultDialect {
28  
29      /**
30       * Example of a query we are building here:
31       *
32       * <code>SELECT * FROM ( SELECT t2.*, ROWNUM as rn FROM ( SELECT t.* FROM PAGES t WHERE <conditions> order by id ) t2) WHERE rn >= 20 AND rownum <= 10;</code>
33       *
34       * Look here for reference: <a href="http://explainextended.com/2009/05/06/oracle-row_number-vs-rownum/">Oracle: ROW_NUMBER vs ROWNUM</a>
35       *
36       * @param tableName name of table
37       * @param subQuery sub query, something like: "name = ? and ssn = ?". It can be blank: "" or null;
38       * @param orderBys collection of order by: "dob desc" - one example
39       * @param limit limit value, -1 if not needed.
40       * @param offset offset value, -1 if not needed.
41       * @return Oracle - specific select query. Here is one example:
42       *
43       * <code>SELECT * FROM ( SELECT t2.*, ROWNUM as rn FROM ( SELECT t.* FROM PAGES t  WHERE <conditions> order by id ) t2) WHERE rn >= 20 AND rownum <= 10;</code>
44       * Can't think of an uglier thing. Shame on you, Oracle.
45       */
46      public String formSelect(String tableName, String subQuery, List<String> orderBys, long limit, long offset) {
47  
48          boolean needLimit = limit != -1;
49          boolean needOffset = offset != -1;
50  
51          offset += 1;//Oracle offset starts with 1, not like MySQL with 0;
52  
53          String fullQuery = needLimit || needOffset ? "SELECT t.* FROM " + tableName + " t " : "SELECT * FROM " + tableName;
54  
55          if(!Util.blank(subQuery)){
56              String where = " WHERE ";
57              //this is only to support findFirst("order by..."), might need to revisit later
58              if(!subQuery.toLowerCase().trim().startsWith("order") && !subQuery.toLowerCase().trim().startsWith("group")){
59                  fullQuery += where;
60              }
61              fullQuery += subQuery;
62          }
63  
64          if(orderBys.size() != 0){
65              fullQuery += " ORDER BY " + Util.join(orderBys, ", ");
66          }
67  
68          String tmp;
69          if(needLimit && needOffset){
70              tmp = "SELECT * FROM ( SELECT t2.*, ROWNUM as ORACLE_ROW_NUMBER FROM ( " + fullQuery + " ) t2) WHERE ";
71              fullQuery = tmp +   "ORACLE_ROW_NUMBER >= " + offset + " AND ROWNUM <= " + limit;
72          }
73          else if(needLimit && !needOffset){
74              tmp = "SELECT * FROM ( SELECT t2.* FROM ( " + fullQuery + " ) t2) WHERE ";
75              fullQuery = tmp + "ROWNUM <= " + limit;            
76          }else if(needOffset){
77              tmp = "SELECT * FROM ( SELECT t2.*, ROWNUM as ORACLE_ROW_NUMBER FROM ( " + fullQuery + " ) t2) WHERE ";
78              fullQuery = tmp +   "ORACLE_ROW_NUMBER >= " + offset;
79          }
80  
81          return fullQuery;
82      }
83  }
