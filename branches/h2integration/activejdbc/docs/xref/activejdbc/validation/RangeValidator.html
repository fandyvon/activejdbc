18  package activejdbc.validation;
19  
20  import activejdbc.Messages;
21  import activejdbc.Model;
22  
23  import java.lang.reflect.Method;
24  import java.math.BigDecimal;
25  import java.util.Locale;
26  
27  public class RangeValidator implements Validator{
28      private String attribute;
29      private Number min, max;
30      private String message;
31  
32      public RangeValidator(String attribute, Number min, Number max){
33          this.attribute = attribute;
34          this.min =  min;
35          this.max = max;
36  
37          if(!min.getClass().equals(max.getClass())){throw new IllegalArgumentException("min and max must be the same type");}
38          message = "value should be within limits: > {0} and < {1}";
39      }
40  
41      public void setMessage(String message) {
42          this.message = message;
43      }
44  
45      public void validate(Model m) {
46          if(m.get(attribute) == null){
47              m.addValidator(attribute, this);
48              return;
49          }
50          Object value = m.get(attribute);
51          if(!value.getClass().equals(max.getClass())){
52              throw new IllegalArgumentException("attribute " + attribute + " type(class) must be the same type as range limits. Min type: "
53                      + min.getClass() + ", Max type: " + max.getClass() + ", Attribute name: " + attribute + ", attribute type: " + value.getClass());
54          }
55  
56          //I hate code in this class, but it is 1:30 AM... :(
57          if(value.getClass().equals(BigDecimal.class) || value.getClass().equals(BigDecimal.class)){
58              try {
59                  Method compareTo = value.getClass().getMethod("compareTo");
60                  if(((Integer)compareTo.invoke(value, min)) == -1 || ((Integer)compareTo.invoke(value, max)) == 1){
61                      m.addValidator(attribute, this);
62                  }
63              } catch (Exception e) {throw new RuntimeException(e);}
64  
65          }else{
66              if(value.getClass().equals(Byte.class)){
67                  Byte v = (Byte)value;
68                  Byte mn = (Byte)min;
69                  Byte mx = (Byte)max;
70                  if(v > mx || v < mn)
71                      m.addValidator(attribute, this);
72              }else if(value.getClass().equals(Double.class)){
73                  Double v = (Double)value;
74                  Double mn = (Double)min;
75                  Double mx = (Double)max;
76                  if(v > mx || v < mn)
77                      m.addValidator(attribute, this);
78              }else if(value.getClass().equals(Float.class)){
79                  Float v = (Float)value;
80                  Float mn = (Float)min;
81                  Float mx = (Float)max;
82                  if(v > mx || v < mn)
83                      m.addValidator(attribute, this);
84              }else if(value.getClass().equals(Integer.class)){
85                  Integer v = (Integer)value;
86                  Integer mn = (Integer)min;
87                  Integer mx = (Integer)max;
88                  if(v > mx || v < mn)
89                      m.addValidator(attribute, this);
90              }else if(value.getClass().equals(Long.class)){
91                  Long v = (Long)value;
92                  Long mn = (Long)min;
93                  Long mx = (Long)max;
94                  if(v > mx || v < mn)
95                      m.addValidator(attribute, this);
96              }else if(value.getClass().equals(Short.class)){
97                  Short v = (Short)value;
98                  Short mn = (Short)min;
99                  Short mx = (Short)max;
100                 if(v > mx || v < mn)
101                     m.addValidator(attribute, this);
102             }
103         }
104     }
105 
106     public String formatMessage(Locale locale, Object ... params) {//params not used
107         return locale != null ? Messages.message(message, locale, min, max) : Messages.message(message, min.toString(), max.toString());         
108     }
