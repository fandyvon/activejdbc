72      /**
73       * Hydrates a this instance of model from a map. Only picks values from a map that match
74       * this instance's attribute names, while ignoring the others.
75       *
76       * @param attributesMap map containing values for this instance.
77       */
78      protected  void hydrate(Map attributesMap) {
79  
80          List<String> attributeNames = getMetaModelLocal().getAttributeNamesSkipId();
81          
82          String idName = getMetaModelLocal().getIdName();
83          Object id = attributesMap.get(idName);
84  
85          if(id != null)
86              attributes.put(idName, id);
87  
88          for (String attrName : attributeNames) {
89  
90              if(attrName.equalsIgnoreCase(getMetaModelLocal().getIdName())){
91                  continue;//skip ID, already set.
92              }
93  
94              Object value = attributesMap.get(attrName.toLowerCase());
95              if (value == null) {
96                  value = attributesMap.get(attrName.toUpperCase());
97              }
98  
99              //it is necessary to cache contents of a clob, because if a clob instance itself s cached, and accessed later,
100             //it will not be able to connect back to that same connection from which it came.
101             //This is only important for cached models. This will allocate a ton of memory if Clobs are large.
102             //Should the Blob behavior be the same?
103             //TODO: write about this in future tutorial
104             if(value != null && value instanceof Clob && getMetaModelLocal().cached() ){
105                 this.attributes.put(attrName.toLowerCase(), Convert.toString(value));
106             }else if(value != null){
107                 this.attributes.put(attrName.toLowerCase(), value);
108             }
109         }
110     }

242     /**
243      * Deletes this record from table.
244      * After deletion, this instance becomes {@link #frozen()} and cannot be used anymore until {@link #thaw()} is called.
245      * <h4>One to many relationships:</h4>
246      * Deletes current model and all one to many associations. This is not a high performance method, as it will
247      * load every row into a model instance before deleting, effectively calling (N + 1) per table queries to the DB, one to select all
248      * the associated records (per table), and one delete statement per record. Use it for small data sets.
249      * It will follow associations of children and their associations too.
250      * <h4>Many to many relationships:</h4>
251      * Deletes current model and will navigate through all many to many relationships
252      * this model has, and clear links to other tables in join tables. This is a high performance call because links are cleared with one
253      * SQL DELETE.
254      * <h4>One to many polymorphic relationship:</h4>
255      * Deletes current model and all polymorphic children. This is a high performance call because links are cleared with one
256      * SQL DELETE. 
257      */
258     public void deleteCascade(){
259         deleteOneToManyChildren();
260         deleteJoinsForManyToMany();
261         deletePolymorphicChildren();
262         delete();
263     }

265     private void deleteJoinsForManyToMany() {
266         List<Association> associations = getMetaModelLocal().getManyToManyAssociations();
267         for(Association association:associations){
268             String join = ((Many2ManyAssociation)association).getJoin();
269             String sourceFK = ((Many2ManyAssociation)association).getSourceFkName();
270             String query = "DELETE FROM " + join + " WHERE " + sourceFK + " = " + getId();
271             new DB(getMetaModelLocal().getDbName()).exec(query);
272         }
273     }

275     private void deletePolymorphicChildren() {
276         List<OneToManyPolymorphicAssociation> polymorphics = getMetaModelLocal().getPolymorphicAssociations();
277         for (OneToManyPolymorphicAssociation association : polymorphics) {            
278             String  target = association.getTarget();
279             String parentType = association.getParentType();
280             String query = "DELETE FROM " + target + " WHERE parent_id = ? AND parent_type = ?";
281             new DB(getMetaModelLocal().getDbName()).exec(query, getId(), parentType);
282         }
283     }

285     private void deleteOneToManyChildren(){
286         List<Association> one2manies = getMetaModelLocal().getOneToManyAssociations();
287         for (Association association : one2manies) {
288             String targetTableName = association.getTarget();
289             Class c = Registry.instance().getModelClass(targetTableName);
290             if(c == null)// this model is probably not defined as a class, but the table exists!
291             {
292                 logger.error("ActiveJDBC WARNING: failed to find a model class for: " + targetTableName + ", maybe model is not defined for this table?" +
293                         " There might be a risk of running into integrity constrain violation if this model is not defined.");
294             }
295             else{
296                 List<Model> dependencies = getAll(c);
297                 for (Model model : dependencies) {
298                     model.deleteCascade();
299                 }
300             }
301         }
302     }

305     /**
306      * Deletes some records from associated table. This method does not follow any associations.
307      * If this model has one to many associations, you might end up with either orphan records in child
308      * tables, or run into integrity constraint violations. However, this method if very efficient as it deletes all records
309      * in one shot, without pre-loading them.
310      * This method also has a side-effect: it will not mark loaded instances corresponding to deleted records as "frozen".
311      * This means that such an instance would allow calling save() and saveIt() methods resulting DB errors, as you
312      * would be attempting to update phantom records. 
313      *
314      *
315      * @param query narrows which records to delete. Example: <pre>"last_name like '%sen%'"</pre>.
316      * @param params   (optional) - list of parameters if a query is parametrized.
317      * @return number od deleted records.
318      */
319     public static int delete(String query, Object... params) {
320         MetaModel metaModel = getMetaModel();
321         int count =  params == null || params.length == 0? new DB(metaModel.getDbName()).exec("DELETE FROM " + metaModel.getTableName() + " WHERE " + query) :
322         new DB(metaModel.getDbName()).exec("DELETE FROM " + metaModel.getTableName() + " WHERE " + query, params);
323         if(metaModel.cached()){
324             QueryCache.instance().purgeTableCache(metaModel.getTableName());
325         }
326         purgeEdges();
327         return count;
328     }

359      * Updates records associated with this model.
360      *
361      * This example :
362      * <pre>
363      *  Employee.update("bonus = ?", "years_at_company > ?", "5", "10");
364      * </pre
365      * In this example, employees who worked for more than 10 years, get a bonus of 5% (not so generous :)).
366      *
367      *
368      * @param updates - what needs to be updated.
369      * @param conditions specifies which records to update. If this argument is <code>null</code>, all records in table will be updated.
370      * In such cases, use a more explicit {@link #updateAll(String, Object...)} method.
371      * @param params list of parameters for both updates and conditions. Applied in the same order as in the arguments,
372      * updates first, then conditions.
373      * @return number of updated records.
374      */
375     public static int update(String updates, String conditions, Object ... params) {
376 
377         //TODO: validate that the number of question marks is the same as number of parameters
378 
379         MetaModel metaModel = getMetaModel();
380         Object []allParams;
381         if(metaModel.hasAttribute("updated_at")){
382             updates = "updated_at = ?, " + updates;
383             allParams = new Object[params.length + 1];
384             System.arraycopy(params, 0, allParams, 1, params.length);
385 
386             allParams[0] = new Timestamp(System.currentTimeMillis());
387         }else{
388             allParams = params;
389         }
390         String sql = "UPDATE " + metaModel.getTableName() + " SET " + updates + ((conditions != null)?" WHERE " + conditions:"");
391         int count = new DB(metaModel.getDbName()).exec(sql, allParams);
392         if(metaModel.cached()){
393             QueryCache.instance().purgeTableCache(metaModel.getTableName());
394         }
395         return count;
396     }

418     /**
419      * Returns all values of the model with all attribute names converted to lower case,
420      * regardless how these names came from DB. This method is a convenience
421      * method for displaying values on web pages.
422      *
423      * <p/>
424      * If {@link activejdbc.LazyList#include(Class[])} method was used, and this
425      * model belongs to a parent (as in many to one relationship), then the parent
426      * will be eagerly loaded and also converted to a map. Parents' maps are keyed in the
427      * returned map by underscored name of a parent model class name.
428      * <p/>
429      * For example, if this model were <code>Address</code>
430      * and a parent is <code>User</code> (and user has many addresses), then the resulting map would
431      * have all the attributes of the current table and another map representing a parent user with a
432      * key "user" in current map.
433      *
434      * @return all values of the model with all attribute names converted to lower case.
435      */
436     public Map<String, Object> toMap(){
437         Map<String, Object> retVal = new HashMap<String, Object>();
438         for (String key : attributes.keySet()) {
439             retVal.put(key.toLowerCase(), attributes.get(key));
440         }
441         for(Class parentClass: cachedParents.keySet()){
442             retVal.put(underscore(shortName(parentClass.getName())), cachedParents.get(parentClass).toMap());
443         }
444 
445         for(Class childClass: cachedChildren.keySet()){
446             List<Model> children = cachedChildren.get(childClass);
447 
448             List<Map> childMaps = new ArrayList<Map>(children.size());
449             for(Model child:children){
450                 childMaps.add(child.toMap());
451             }
452             retVal.put(pluralize(underscore(shortName(childClass.getName()))), childMaps);
453         }
454         return retVal;
455     }

457     /**
458      * Generates a XML document from content of this model.
459      *
460      * @param spaces by how many spaces to indent.
461      * @param declaration tru to include XML declaration at the top
462      * @param attrs list of attributes to include. No arguments == include all attributes.
463      * @return generated XML.
464      */
465     public String toXml(int spaces, boolean declaration, String ... attrs){
466 
467 
468         Map<String, Object> modelMap = toMap();
469 
470         String indent = "";
471         for(int i = 0; i < spaces; i++)
472             indent += " ";
473 
474         List<String> attrList = Arrays.asList(attrs);
475 
476         StringWriter sw = new StringWriter();
477 
478         if(declaration) sw.write("<?xml version=\"1.0\" encoding=\"UTF-8\" ?>" + (spaces > 0?"\n":""));
479 
480         String topTag = Inflector.underscore(getClass().getSimpleName());
481 
482         sw.write(indent + "<" + topTag + ">" + (spaces > 0?"\n":""));
483 
484 
485         for(String name: modelMap.keySet()){
486             Object value  = modelMap.get(name);
487             if((attrList.contains(name) || attrs.length == 0) && !(value instanceof List)){
488                 sw.write(indent + indent + "<" + name + ">" + value + "</" + name + ">" + (spaces > 0?"\n":""));
489             }else if (value instanceof List){
490                 List<Map> children = (List<Map>)value;
491                 sw.write(indent + indent + "<" + name + ">" + (spaces > 0?"\n":""));
492                 for(Map child: children){
493                     sw.write(indent + indent + indent + "<" + Inflector.singularize(name) + ">" + (spaces > 0?"\n":""));
494                     for(Object childKey: child.keySet()){
495                         sw.write(indent + indent + indent + indent + "<" + childKey + ">" + child.get(childKey)+ "</" + childKey +">" + (spaces > 0?"\n":""));
496                     }
497                     sw.write(indent + indent + indent + "</" + Inflector.singularize(name) + ">" + (spaces > 0?"\n":""));
498                 }
499                 sw.write(indent + indent + "</" + name + ">" + (spaces > 0?"\n":""));
500             }
501         }
502         sw.write(indent + "</" + topTag + ">"+ (spaces > 0?"\n":""));
503         return sw.toString();
504     }

510     /**
511      * Returns parent of this model, assuming that this table represents a child.
512      *
513      * @param parentClass   class of a parent model.
514      * @return instance of a parent of this instance in the "belongs to"  relationship.
515      */
516     public <T extends Model> T parent(Class<T> parentClass) {
517 
518         T cachedParent = (T)cachedParents.get(parentClass);
519         if(cachedParent != null){
520             return cachedParent;
521         }
522         MetaModel parentMM = Registry.instance().getMetaModel(parentClass);
523         String parentTable = parentMM.getTableName();
524 
525         BelongsToAssociation ass = (BelongsToAssociation)getMetaModelLocal().getAssociationForTarget(parentTable, BelongsToAssociation.class);
526         BelongsToPolymorphicAssociation assP = (BelongsToPolymorphicAssociation)getMetaModelLocal()
527                 .getAssociationForTarget(parentTable, BelongsToPolymorphicAssociation.class);
528 
529         String fkValue;
530         String fkName;
531         if(ass != null){
532             fkValue = getString(ass.getFkName());
533             fkName = ass.getFkName();
534         }else if(assP != null){
535             fkValue = getString("parent_id");
536             fkName = "parent_id";
537             MetaModel trueMM = Registry.instance().getMetaModelByClassName(getString("parent_type"));
538             if(trueMM != parentMM)
539                 throw new IllegalArgumentException("Wrong parent: '" + parentClass + "'. Actual parent type of this record is: '" + getString("parent_type") + "'");
540         }else{
541             throw new DBException("there is no association with table: " + parentTable);
542         }
543 
544         if(fkValue == null){
545             throw new InternalException("Attribute:  "  + fkName + " is null, cannot determine parent.");
546         }
547         String parentIdName = parentMM.getIdName();
548         String query = getMetaModelLocal().getDialect().selectStarParametrized(parentTable, parentIdName);
549 
550         T parent;
551         if(parentMM.cached()){
552             parent = (T)QueryCache.instance().getItem(parentTable, query, new Object[]{fkValue});
553             if(parent != null){
554                 return parent;
555             }
556         }
557 
558         List<Map> results = new DB(getMetaModelLocal().getDbName()).findAll(query, Integer.parseInt(fkValue));
559         //expect only one result here
560         if (results.size() == 0) { //ths could be covered by referential integrity constraint
561             return null;
562         } else {
563             try {
564                 parent = parentClass.newInstance();
565                 parent.hydrate(results.get(0));
566                 if(parentMM.cached()){
567                     QueryCache.instance().addItem(parentTable, query, new Object[]{fkValue}, parent);
568                 }
569                 return parent;
570             } catch (Exception e) {
571                 throw new InitException(e.getMessage(), e);
572             }
573         }
574     }

594     /**
595      * Sets a parent on this instance. Basically this sets a correct value of a foreign key in a
596      * parent/child relationship. This only works for one to many and polymorphic associations.
597      *
598      * @param parent potential parent of this instance. Its ID value must not be null.
599      */
600     public void setParent(Model parent) {
601         if (parent == null || parent.getId() == null) {
602             throw new IllegalArgumentException("parent cannot ne null and parent ID cannot be null");
603         }
604         List<Association> associations = getMetaModelLocal().getAssociations();
605         for (Association association : associations) {
606             if (association instanceof BelongsToAssociation && association.getTarget().equals(parent.getMetaModelLocal().getTableName())) {
607                 set(((BelongsToAssociation)association).getFkName(), parent.getId());
608                 return;
609             }
610             if(association instanceof BelongsToPolymorphicAssociation && association.getTarget().equals(parent.getMetaModelLocal().getTableName())){
611                 set("parent_id", parent.getId());
612                 set("parent_type", ((BelongsToPolymorphicAssociation)association).getParentType());
613                 return;
614             }
615         }
616         throw new IllegalArgumentException("Class: " + parent.getClass() + " is not associated with " + this.getClass()
617                 + ", list of existing associations: \n" + Util.join(getMetaModelLocal().getAssociations(), "\n"));
618     }

668      * Returns a value for attribute. Besides returning direct attributes of this model, this method is also
669      * aware of relationships and can return collections based on naming conventions. For example, if a model User has a
670      * one to many relationship with a model Address, then the following code will work:
671      * <pre>
672      * Address address = ...;
673      * User user = (User)address.get("user");
674      * </pre>
675      * Conversely, this will also work:
676      * <pre> 
677      * List<Address> addresses = (List<Address>)user.get("addresses");
678      * </pre>
679      *
680      * The same would also work for many to many relationships:
681      * <pre>
682      * List<Doctor> doctors = (List<Doctor>)patient.get("doctors");
683      * ...
684      * List<Patient> patients = (List<Patient>)doctor.get("patients");
685      * </pre>
686      *
687      * This methods will try to infer a name if a table by using {@link javalite.common.Inflector} to try to
688      * convert it to singular and them plural form, an attempting to see if this model has an appropriate relationship
689      * with another model, if one is found. This method of finding of relationships is best used in templating
690      * technologies, such as JSPs. For standard cases, please use {@link #parent(Class)}, and {@link #getAll(Class)}.
691      *
692      *
693      * @param attribute name of attribute.
694      * @return value for attribute.
695      */
696     public Object get(String attribute) {
697         if(frozen) throw new FrozenException(this);
698 
699         if(attribute == null) throw new IllegalArgumentException("attribute cannot be null");
700 
701         // NOTE: this is a workaround for JSP pages. JSTL in cases ${item.id} does not call the getId() method, instead
702         //calls item.get("id"), considering that this is a map only!        
703         if(attribute.equalsIgnoreCase("id")){
704             String idName = getMetaModelLocal().getIdName();
705             return attributes.get(idName.toLowerCase());
706         }
707 
708         Object returnValue;
709         String attributeName = attribute.toLowerCase();
710         if((returnValue = attributes.get(attributeName.toLowerCase())) != null){
711             return returnValue;
712         }else if((returnValue = tryParent(attributeName)) != null){
713             return returnValue;
714         }else if((returnValue = tryPolymorphicParent(attributeName)) != null){
715             return returnValue;
716         }else if((returnValue = tryChildren(attributeName)) != null){
717             return returnValue;
718         }else if((returnValue = tryPolymorphicChildren(attributeName)) != null){
719             return returnValue;
720         }else if((returnValue = tryOther(attributeName)) != null){
721             return returnValue;
722         }else{
723             getMetaModelLocal().checkAttributeOrAssociation(attributeName);
724             return null;
725         }
726     }

986      *

996     private <T extends Model> LazyList<T> get(String targetTable, String criteria, Object ...params) {
997         //TODO: interesting thought: is it possible to have two associations of the same name, one to many and many to many? For now, say no.
998 
999         OneToManyAssociation oneToManyAssociation = (OneToManyAssociation)getMetaModelLocal().getAssociationForTarget(targetTable, OneToManyAssociation.class);
1000         Many2ManyAssociation manyToManyAssociation = (Many2ManyAssociation)getMetaModelLocal().getAssociationForTarget(targetTable, Many2ManyAssociation.class);
1001 
1002         OneToManyPolymorphicAssociation oneToManyPolymorphicAssociation = (OneToManyPolymorphicAssociation)getMetaModelLocal().getAssociationForTarget(targetTable, OneToManyPolymorphicAssociation.class);
1003 
1004         String additionalCriteria =  criteria != null? " AND ( " + criteria + " ) " : "";
1005         String subQuery;
1006         if (oneToManyAssociation != null) {
1007             subQuery = oneToManyAssociation.getFkName() + " = " + getId() + additionalCriteria;
1008         } else if (manyToManyAssociation != null) {
1009             String targetId = Registry.instance().getMetaModel(targetTable).getIdName();
1010 
1011             subQuery = targetId + " IN ( SELECT " +
1012                 manyToManyAssociation.getTargetFkName() + " FROM " + manyToManyAssociation.getJoin() + " WHERE " +
1013                 manyToManyAssociation.getSourceFkName() + " = " + getId() + additionalCriteria + ")"   ;

1017         } else {
1018             throw new NotAssociatedException(getMetaModelLocal().getTableName(), targetTable);
1019         }
1020         return new LazyList<T>(subQuery, params,Registry.instance().getMetaModel(targetTable));
