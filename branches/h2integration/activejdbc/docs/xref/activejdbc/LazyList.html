124     /**
125      * Use this method includes associated objects. This method will eagerly load associated models of
126      * models selected by the query. For instance, if there are models <code>Author</code>, <code>Post</code>
127      * and <code>Comment</code>, where <code>Author</code> has many <code>Post</code>s and <code> Post</code>
128      * has many <code>Comment</code>s, then this query:
129      * <pre>
130      * List<Post> todayPosts = Post.where("post_date = ?", today).include(Author.class, Comment.class);
131      * </pre>
132      * will generate only three queries to database - one per model. All the dependencies (includes) will be
133      * eagerly loaded, and iteration via the <code>todayPosts</code> list will not generate any more queries,
134      * even when a post author and comments are requested. Use this with caution as this method can allocate
135      * a lot of memory (obviously).
136      *
137      * <p/>
138      *
139      * This method will not follow relationships of related models, but rather only relationships of the current
140      * one.  
141      *
142      * @param classes list of dependent classes. These classes represent models with which a current model has a
143      * relationship.
144      * @return instance of this <code>LazyList</code>
145      */

148     public <E extends Model>  LazyList<E>  include(Class<? extends Model> ... classes){
149         if(includes.size() != 0) throw new IllegalArgumentException("Can't call include() more than once!");
150 
151         //lets cache included classes and associations for future processing.
152         for(Class includeClass: classes){
153             String table = Registry.instance().getTableName(includeClass);
154             Association association = metaModel.getAssociationForTarget(table);
155             if(association == null){
156                 throw new IllegalArgumentException("this model is not associated with " + includeClass);
157             }
158 
159             includes.put(includeClass, association);
160         }
161 
162         return (LazyList<E>)this;
163     }

179     /**
180      * Generates a XML document from content of this list.
181      *
182      * @param spaces by how many spaces to indent.
183      * @param declaration true to include XML declaration at the top
184      * @param attrs list of attributes to include. No arguments == include all attributes.
185      * @return generated XML.
186      */
187     public String toXml(int spaces, boolean declaration, String ... attrs){
188         String topNode = Inflector.pluralize(Inflector.underscore(metaModel.getModelClass().getSimpleName()));
189 
190         hydrate();
191 
192         StringWriter sw = new StringWriter();
193         if(declaration)
194             sw.write("<?xml version=\"1.0\" encoding=\"UTF-8\" ?>" + (spaces > 0?"\n":""));
195 
196         sw.write("<"  + topNode + ">" + (spaces > 0 ? "\n":""));
197         for (T t : delegate) {
198             sw.write(t.toXml(spaces, false, attrs));
199         }
200         sw.write("</"  + topNode + ">" + (spaces > 0 ? "\n":""));
201         return sw.toString();
202     }

207      * <code> Person.find("name = ?", "Smith").load();</code>.
208      * It is not possible to call other methods after load(). The load() method should be the last to be called in the chain:

212      * @return fully loaded list.

215     public <E extends Model>  LazyList<E>  load(){
216         
217         if(hydrated) throw new DBException("load() must be the last on the chain of methods");
218 
219         hydrate();
220         return (LazyList<E>) this;
221     }

223     protected void hydrate(){
224 
225         if(hydrated) return;
226 
227         String subQuery = Util.join(subQueries.toArray(new String[]{}), " ");
228 
229         String sql = fullQuery != null? fullQuery :
230                 Registry.instance().getConfiguration().getDialect(metaModel).formSelect(metaModel.getTableName(), subQuery,
231                         orderBys, limit, offset);
232 
233         if(metaModel.cached()){        
234             ArrayList<T> cached = (ArrayList<T>) QueryCache.instance().getItem(metaModel.getTableName(), sql, params);
235             if(cached != null){
236                 delegate = cached;
237                 return;
238             }
239         }
240 
241         long start = System.currentTimeMillis();
242         new DB(metaModel.getDbName()).find(sql, params).with(new RowListenerAdapter() {
243             public void onNext(Map<String, Object> rowMap) {
244                 delegate.add((T) Model.instance(rowMap, metaModel));
245             }
246         });
247         LogFilter.logQuery(logger, sql, params, start);
248         if(metaModel.cached()){
249             QueryCache.instance().addItem(metaModel.getTableName(), sql, params, delegate);
250         }
251         hydrated = true;
252         processIncludes();        
253     }

255     private void processIncludes(){
256 
257         for(Class includedClass: includes.keySet()){            
258             Association association = includes.get(includedClass);
259             if(association instanceof BelongsToAssociation){
260                 processParent((BelongsToAssociation)association, includedClass);
261             }else if(association instanceof OneToManyAssociation){
262                 processChildren((OneToManyAssociation)association, includedClass);
263             }else if(association instanceof Many2ManyAssociation){
264                 processOther((Many2ManyAssociation)association, includedClass);
265             }
266         }        
267     }

269     private void processParent(BelongsToAssociation association, Class parentClass) {
270 
271         if(delegate.size() == 0){//no need to process parents if no models selected.
272             return;
273         }
274 
275         final MetaModel parentMM = Registry.instance().getMetaModel(parentClass);
276         final Map<Object, Model> parentsHasByIds = new HashMap<Object, Model>();
277 
278         String fkName = association.getFkName();
279 
280         //need to remove duplicates because more than one child can belong to the same parent. 
281         List parentIds = collect(fkName);
282         ArrayList noDuplicateList = new ArrayList(new HashSet(parentIds));
283 
284         for(Model parent: new LazyList<Model>(parentMM.getIdName() + " IN (" + Util.join(noDuplicateList, ", ") + ")", null, parentMM)){
285                parentsHasByIds.put(parent.getId(), parent);
286         }
287 
288         //now that we have the parents in the has, we need to distribute them into list of children that are
289         //stored in the delegate.
290         for(Model child: delegate){
291             Object fk = child.get(fkName);
292             Model parent = parentsHasByIds.get(fk);
293             child.setCachedParent(parent);
294         }
295     }

297     /**
298      * Collects values from a result set that correspond to a column name.
299      * For example, if a list contains collection of <code>Person</code> models, then
300      * you can collect first names like this:
301      * <pre>
302      * List firstNames = Person.findAll().collect("first_name");
303      * </pre>
304      * provided that the corresponding table has a column <code>first_name</code>.
305      * <p/><p/>
306      * Bare in mind, that if all you need is a one column data, this method of getting it is not
307      * the most efficient (because since you are using a model, you will query all columns from a table,
308      * but will use only one). In these cases, you might want to consider {@link Base#firstColumn(String, Object...)} and
309      * {@link DB#firstColumn(String, Object...)}.
310      *
311      * @param columnName name of column to collect.
312      * @return list of collected values for a column.
313      */
314     public List collect(String columnName){
315         hydrate();
316         List results = new ArrayList();
317         for(Model model: delegate){
318             results.add(model.get(columnName));
319         }
320 
321         return results;
322     }

324     private void processChildren(OneToManyAssociation association, Class childClass) {
325 
326         if(delegate.size() == 0){//no need to process children if no models selected.
327             return;
328         }
329 
330         final MetaModel childMM = Registry.instance().getMetaModel(childClass);
331         final String fkName = association.getFkName();
332 
333         final Map<Object, List<Model>> childrenByParentId = new HashMap<Object, List<Model>>();
334 
335         List ids = collect(metaModel.getIdName());
336 
337         for(Model child: new LazyList<Model>(fkName + " IN (" + Util.join(ids, ", ") + ")" , null, childMM).orderBy(childMM.getIdName())){
338              if(childrenByParentId.get(child.get(fkName)) == null){
339                     childrenByParentId.put(child.get(fkName), new SuperLazyList<Model>());
340              }
341             childrenByParentId.get(child.get(fkName)).add(child);
342         }
343 
344         for(T parent : delegate){
345             List<Model> children = childrenByParentId.get(parent.getId());
346             if(children != null){
347                 parent.setChildren(childClass, children);
348             }
349         }
350     }

352     private void processOther(Many2ManyAssociation association, Class childClass) {
353         if(delegate.size() == 0){//no need to process other if no models selected.
354             return;
355         }
356 
357         final MetaModel childMM = Registry.instance().getMetaModel(childClass);
358         final Map<Object, List<Model>> childrenByParentId = new HashMap<Object, List<Model>>();
359 
360         List ids = collect(metaModel.getIdName());
361         
362         String sql =  "SELECT " + association.getTarget() + ".*, t." + association.getSourceFkName() + " AS the_parent_record_id FROM " + association.getTarget() +
363         " INNER JOIN " + association.getJoin() + " t ON " + association.getTarget() + " .id = t." + association.getTargetFkName() + " WHERE (t." + association.getSourceFkName()
364                 + "  IN (" + Util.join(ids, ", ") + "))";
365 
366 
367         List<Map> childResults = new DB(childMM.getDbName()).findAll(sql);
368 
369         for(Map res: childResults){
370             Model child = Model.instance(res, childMM);
371             Object parentId = res.get("the_parent_record_id");
372             if(childrenByParentId.get(parentId) == null){
373                     childrenByParentId.put(parentId, new SuperLazyList<Model>());
374              }
375             childrenByParentId.get(parentId).add(child);
376         }
377         
378         for(T parent : delegate){
379             List<Model> children = childrenByParentId.get(parent.getId());
380             if(children != null){
381                 parent.setChildren(childClass, children);
382             }
383         }
384     }
